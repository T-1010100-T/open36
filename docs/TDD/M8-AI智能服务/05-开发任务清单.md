  ```python
  async def call_internal(service_url, method, path, headers_extra=None, **kwargs):
      async with httpx.AsyncClient() as client:
          resp = await client.request(method, f"{service_url}{path}",
              headers={"X-Internal-API-Key": settings.INTERNAL_API_KEY, **(headers_extra or {})},
              timeout=30.0, **kwargs)
          resp.raise_for_status()
          return resp.json()
  ```
